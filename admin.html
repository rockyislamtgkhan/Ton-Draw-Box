fetch("/withdraw-requests") .then(res => res.json()) .then(data => { const table = document.getElementById("withdraw-table"); data.reverse().forEach(req => { const row = document.createElement("tr"); row.innerHTML = ` ${req.userId} @${req.username} ${req.wallet} ${req.amount} TON ${new Date(req.time).toLocaleString()} `; table.appendChild(row); }); });
